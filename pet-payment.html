// ==================== Pet Payment 优化版（方便后期接链上价格） ==================== const PAYMENT_CONFIG = { // 暂时使用，等 PAWLY Token 上线后，改成从链上获取真实价格 pawlyPerUsdc: 5, // 临时比例：5 PAWLY = 1 USDC }; // 打开弹窗 function openPetPaymentModal() { document.getElementById('petPaymentModal').style.display = 'flex'; document.getElementById('paymentAmount').value = ''; document.getElementById('usdcAmount').innerText = '计算中...'; document.getElementById('myrAmount').innerText = '获取汇率中...'; // 打开时获取最新汇率 fetchLatestRates(); } // 关闭弹窗 function closePetPaymentModal() { document.getElementById('petPaymentModal').style.display = 'none'; } // 获取实时 USD → MYR 汇率 async function fetchLatestRates() { try { const response = await fetch('https://api.exchangerate-api.com/v4/latest/USD'); if (!response.ok) { throw new Error('汇率接口请求失败'); } const data = await response.json(); if (!data.rates || !data.rates.MYR) { throw new Error('无法获取 MYR 汇率'); } window.currentMyrRate = data.rates.MYR; console.log('✅ 成功获取今日汇率 USD → MYR:', window.currentMyrRate); // 重新计算一次 calculatePayment(); } catch (error) { console.error('获取汇率失败:', error); window.currentMyrRate = null; document.getElementById('usdcAmount').innerText = '—'; document.getElementById('myrAmount').innerText = '汇率获取失败，请稍后重试'; } } // 实时计算 function calculatePayment() { const amount = parseFloat(document.getElementById('paymentAmount').value) || 0; if (amount <= 0) { document.getElementById('usdcAmount').innerText = '0.00 USDC'; document.getElementById('myrAmount').innerText = 'RM 0.00'; return; } // 使用配置的 PAWLY 比例（后期可替换为链上真实价格） const usdc = amount / PAYMENT_CONFIG.pawlyPerUsdc; document.getElementById('usdcAmount').innerText = usdc.toFixed(2) + ' USDC'; // 如果汇率获取成功才显示 MYR if (window.currentMyrRate) { const myr = usdc * window.currentMyrRate; document.getElementById('myrAmount').innerText = 'RM ' + myr.toFixed(2); } else { document.getElementById('myrAmount').innerText = '汇率获取中...'; } } // 确认支付（目前模拟） function confirmPetPayment() { const amount = parseFloat(document.getElementById('paymentAmount').value); if (!amount || amount <= 0) { alert('请输入有效的支付金额\nPlease enter a valid amount'); return; } if (!window.currentMyrRate) { alert('汇率尚未获取成功，请稍后重试\nExchange rate not available, please try again later'); return; } const usdc = (amount / PAYMENT_CONFIG.pawlyPerUsdc).toFixed(2); const myr = (usdc * window.currentMyrRate).toFixed(2); alert(`✅ 支付确认成功！\n\n支付金额：${amount} PAWLY\n≈ ${usdc} USDC\n≈ RM ${myr}\n\n（目前为模拟支付，正式链上支付将在代币上线后开放）\n\n✅ Payment confirmed!\nThis is a simulated payment. Real on-chain payment will be available after token launch.`); closePetPaymentModal(); }
